SELECT  I.SUMMARY, I.DESCRIPTION, L.NAME AS LIST 
FROM ISSUES I JOIN ISSUESLISTS L ON L.ID = I.ISSUESLIST_ID;

SELECT  I.SUMMARY, I.DESCRIPTION, U.FIRSTNAME AS USERNAME, U.LASTNAME AS ""
FROM ISSUES I JOIN USERS U ON U.ID = I.ISSUESLIST_ID;

SELECT U.FIRSTNAME, U.LASTNAME, 
COUNT(*) AS ISSUES_LEFT
FROM ISSUES I JOIN USERS U ON U.ID = I.ISSUESLIST_ID
GROUP BY U.ID;
